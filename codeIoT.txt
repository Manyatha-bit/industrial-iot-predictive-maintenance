#include <WiFi.h>
#include <Wire.h>
#include <MPU6050.h>
#include "ThingSpeak.h"

// ---------------- WIFI ----------------
const char* ssid = "vivo Y56 5G";
const char* password = "0123456789";
const char* apiKey = "V6DHOWGZ9PMRCDIP";
unsigned long channelID = 3363441;

WiFiClient client;
MPU6050 mpu;

// ---------------- PIN CONFIG ----------------
#define CURRENT_PIN 36   // ACS712
#define TEMP_PIN 34      // LM35

#define IN1 26
#define IN2 27
#define ENA 25

#define LED 13

// ---------------- THRESHOLDS ----------------
#define TEMP_THRESHOLD  40.0
#define CURRENT_THRESHOLD 1.2

// ---------------- VARIABLES ----------------
float currentA = 0.0;
float temperatureC = 0.0;
float vibration = 0.0;

// Auto calibration
float zeroCurrentVoltage = 0;

// ============================================================
// SETUP
// ============================================================
void setup() {
  Serial.begin(115200);

  pinMode(IN1, OUTPUT);
  pinMode(IN2, OUTPUT);
  pinMode(ENA, OUTPUT);
  pinMode(LED, OUTPUT);

  digitalWrite(LED, LOW);

  // 🔥 START MOTOR
  digitalWrite(IN1, HIGH);
  digitalWrite(IN2, LOW);
  analogWrite(ENA, 200);

  // WiFi
  Serial.print("Connecting WiFi...");
  WiFi.begin(ssid, password);

  while (WiFi.status() != WL_CONNECTED) {
    delay(500);
    Serial.print(".");
  }

  Serial.println("\nWiFi Connected");
  Serial.print("IP Address: ");
  Serial.println(WiFi.localIP());

  ThingSpeak.begin(client);

  // I2C
  Wire.begin();

  // MPU6050
  mpu.initialize();
  if (mpu.testConnection())
    Serial.println("MPU OK");
  else
    Serial.println("MPU FAIL");

  // 🔥 CURRENT SENSOR CALIBRATION
  Serial.println("Calibrating current sensor...");
  float sum = 0;

  for (int i = 0; i < 100; i++) {
    sum += analogRead(CURRENT_PIN);
    delay(10);
  }

  float avg = sum / 100.0;
  zeroCurrentVoltage = avg * (3.3 / 4095.0);

  Serial.print("Zero Current Voltage: ");
  Serial.println(zeroCurrentVoltage);

  Serial.println("System Ready\n");
}

// ============================================================
// LOOP
// ============================================================
void loop() {

  // -------- TEMPERATURE --------
  int rawTemp = analogRead(TEMP_PIN);
  float tempVoltage = rawTemp * (3.3 / 4095.0);
  temperatureC = tempVoltage * 100.0;

  // -------- CURRENT --------
  int rawCurrent = analogRead(CURRENT_PIN);
  float voltage = rawCurrent * (3.3 / 4095.0);
  currentA = (voltage - zeroCurrentVoltage) / 0.1;

  // -------- VIBRATION --------
  int16_t ax, ay, az;
  mpu.getAcceleration(&ax, &ay, &az);
  vibration = abs(ax) + abs(ay) + abs(az);

  // -------- FAULT LOGIC --------
  bool fault = false;

  if (temperatureC > TEMP_THRESHOLD) fault = true;
  if (currentA > CURRENT_THRESHOLD) fault = true;

  // -------- LED ALERT --------
  digitalWrite(LED, fault ? HIGH : LOW);

  // -------- MOTOR PROTECTION --------
  if (fault) {
    digitalWrite(IN1, LOW);
    digitalWrite(IN2, LOW);
    analogWrite(ENA, 0);

    Serial.println("⚠️ Motor Stopped! LED ON");
    Serial.println("⚠️ FAULT DETECTED: Over Current or High Temperature!");
  }

  // -------- SERIAL OUTPUT --------
  Serial.println("------ DATA ------");

  Serial.print("Temp ADC: ");
  Serial.println(rawTemp);
  Serial.print("Temp: ");
  Serial.print(temperatureC);
  Serial.println(" C");

  Serial.print("Current ADC: ");
  Serial.println(rawCurrent);
  Serial.print("Current: ");
  Serial.print(currentA);
  Serial.println(" A");

  Serial.print("Vibration: ");
  Serial.println(vibration);

  Serial.print("Fault Status: ");
  Serial.println(fault ? "FAULT DETECTED!" : "NORMAL");

  Serial.println("------------------");

  // -------- THINGSPEAK --------
  ThingSpeak.setField(1, currentA);
  ThingSpeak.setField(2, temperatureC);
  ThingSpeak.setField(3, vibration);
  ThingSpeak.setField(4, fault ? 1 : 0);

  int result = ThingSpeak.writeFields(channelID, apiKey);

  if (result == 200)
    Serial.println("Data sent!");
  else
    Serial.println("Error: " + String(result));

  Serial.println();

  delay(15000);  // ThingSpeak limit
}
